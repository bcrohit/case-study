    \item Cross-validation for robust evaluation
    \item Hyperparameter tuning for tree depth
    \item Temporal feature engineering:
    \begin{itemize}
        \item hour
        \item weekday
        \item month
        \item season

\begin{frame}{Classification Extension}

\setbeamercolor{item}{fg=tugreen}
\begin{itemize}
    \item Regression predicts continuous BP values
    \item Classification predicts elevated BP categories
    \item Systolic high BP:
    \[
    BP_{sys} \geq 140
    \]
    \item Diastolic high BP:
    \[
    BP_{dia} \geq 90
    \]
    \item Weighted Random Forest used to handle imbalance
\end{itemize}

\end{frame}
