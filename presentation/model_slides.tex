\begin{frame}{Updated Modelling Pipeline}

\setbeamercolor{item}{fg=tugreen}
\begin{itemize}
    \item Train-test split was stratified by age groups
    \item Random Forest depth was tuned to reduce overfitting
    \item Cross-validation was used to check model stability
    \item New temporal variables:
    \begin{itemize}
        \item hour, weekday, month, season
    \end{itemize}
    \item External weather variables:
    \begin{itemize}
        \item temperature and humidity, matched by measurement hour
    \end{itemize}
\end{itemize}

\end{frame}
%==================== SLIDE ====================
\begin{frame}{External Weather Data}

\setbeamercolor{item}{fg=tugreen}
\begin{itemize}
    \item Hourly weather data was downloaded from the Open-Meteo Historical Weather API
    \item Location used: Graz, Austria
    \item Time period matched to the health measurements in 2006
    \item Variables added:
    \begin{itemize}
        \item temperature at 2m
        \item relative humidity at 2m
    \end{itemize}
    \item Weather timestamps were matched to measurement timestamps rounded to the nearest hour
\end{itemize}

\vspace{0.2cm}

\small
Source: Open-Meteo Historical Weather API

\end{frame}

\begin{frame}{Classification Extension}

\setbeamercolor{item}{fg=tugreen}
\begin{itemize}
    \item Regression predicts exact BP values
    \item Classification predicts elevated BP categories
    \item Systolic high BP:
    \[
    BP_{sys} \geq 140
    \]
    \item Diastolic high BP:
    \[
    BP_{dia} \geq 90
    \]
    \item Weighted Random Forest was used because high BP cases were less frequent
\end{itemize}

\end{frame}

\begin{frame}{Current Takeaways and Open Questions}

\setbeamercolor{item}{fg=tugreen}
\begin{itemize}
    \item Temporal and weather features improved regression performance
    \item The model captures population-level trends better than individual values
    \item Classification gives a useful risk-category perspective
    \item Threshold choice strongly affects false positives and false negatives
    \item Open question: should final focus be regression performance, classification, or model interpretation?
\end{itemize}
